The purpose of this web site is for colleges and universities to view as part of the application review. This site was designed and built by me for this purpose. This web site is made using standard HTML, CSS, and javascript, with the w3.css CSS library. It is hosted using Microsoft Azure Sites service, and the source code is stored on a public gitHub repository here. Here is a copy of the source code as of 1/3/2018, but please look at the gitHub page for the most recent copy of the source code.

<script>
var slideIndex = 1;
showDivs(slideIndex);
function plusDivs(n) {
showDivs(slideIndex += n);
}
function showDivs(n) {
var i;
var x = document.getElementsByClassName("slideshow");
if (n > x.length) { slideIndex = 1 }
if (n < 1) { slideIndex = x.length }
for (i = 0; i < x.length; i++) {
x[i].style.display = "none";
}
x[slideIndex - 1].style.display = "block";
}
</script>
